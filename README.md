# ⚡ AGENTIC CORE — Multi-Agent Orchestration Pattern  [@vidorc]
from langgraph.graph import StateGraph, END
from langchain_core.messages import HumanMessage, AIMessage
from typing import TypedDict

class AgentState(TypedDict):
    messages : list
    context  : str
    iteration: int
    next     : str

# ── Build autonomous reasoning loop ──────────────────────────────
graph = StateGraph(AgentState)
graph.add_node("planner",   planner_agent)    # Decompose → Plan
graph.add_node("executor",  executor_agent)   # Act → Retrieve
graph.add_node("reflector", reflection_agent) # Evaluate → Route

graph.add_conditional_edges("reflector", route_decision, {
    "continue": "planner",
    "finish":    END,
})

# 🚀 Deploy → Think → Act → Reflect → Repeat  ∞
agent = graph.compile(checkpointer=memory)
